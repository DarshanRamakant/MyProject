CREATE DATABASE IIITBSOCIALNETWORK;

CREATE USER 'IITBJAVA'@'localhost' IDENTIFIED BY 'IIITBJAVA';

GRANT ALL ON IIITBSOCIALNETWORK TO ‘IITBJAVA’;

CREATE TABLE USER_LOGIN (
				USER_ID INTEGER,
				PASSWORD VARCHAR); 

CREATE TABLE USER_ID_MAP (
				USER_ID INTEGER NOT NULL AUTO_INCREMENT, 
				ROLL_NO VARCHAR, 
				EMAILID VARCHAR,
				PRIMARY KEY (USER_ID)); 

CREATE TABLE USER_PROF_DETAILS (
			USER_ID INTEGER, 
			USER_NAME VARCHAR, 
			PHOTO LONGBLOB,
			GENDER VARCHAR,
			COURSE ID INTEGER,
			PHONE NO INTEGER, 
			FOREIGN KEY (USER_ID)
			      REFERENCES USER_ID_MAP(USER_ID)
			); 

CREATE TABLE USER_INTERESTS ( 
				USER_ID INTEGER, 
				INTEGER INTEREST_ID
				FOREIGN KEY (USER_ID)
				      REFERENCES USER_ID_MAP(USER_ID)
				); 

CREATE TABLE USER_POST_DETAILS ( 
				USER_ID INTEGER,
				POST_CONTENT VARCHAR, 
				TIMESTAMP DATETIME,
				FOREIGN KEY (USER_ID)
      					REFERENCES USER_ID_MAP(USER_ID)
				);

CREATE TABLE USER_FRND_MAP( 
				USER_ID INTEGER, 
				FRND_ID INTEGER, 
				FOREIGN KEY (USER_ID)
				      REFERENCES USER_ID_MAP(USER_ID),
				FOREIGN KEY (FRND_ID)
				      REFERENCES USER_ID_MAP(USER_ID),
			);


CREATE TABLE INTEREST ( 
				INTEGER INTEREST_ID NOT NULL AUTO_INCREMENT, 
				VARCHAR INTEREST_DESC 
			);

CREATE TABLE COURSE ( 
				INTEGER COURSE_ID NOT NULL AUTO_INCREMENT, 
				VARCHAR COURSE_DESC 
			);
